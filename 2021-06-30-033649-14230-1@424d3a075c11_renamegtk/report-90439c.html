89/* Punycode bias adaptation algorithm, RFC 3492 section 6.1 */
90static guint
91adapt (guint delta,
92 guint numpoints,
93 gboolean firsttime)
94{
95 guint k;
96
97 delta = firsttime ? delta / PUNYCODE_DAMP700 : delta / 2;
98 delta += delta / numpoints;
99
100 k = 0;
101 while (delta > ((PUNYCODE_BASE36 - PUNYCODE_TMIN1) * PUNYCODE_TMAX26) / 2)
102 {
103 delta /= PUNYCODE_BASE36 - PUNYCODE_TMIN1;
104 k += PUNYCODE_BASE36;
105 }
106
107 return k + ((PUNYCODE_BASE36 - PUNYCODE_TMIN1 + 1) * delta /
108 (delta + PUNYCODE_SKEW38));
109}

312 * the sub-operations when they would just be equivalent to a
313 * g_strdup().

315
316 /* Remove presentation-only characters */
317 name = remove_junk (hostname, len);
318 if (name)

321 len = -1;
322 }

324 name = (gchar *)hostname;
325
326 /* Convert to lowercase */
327 if (contains_uppercase_letters (name, len))
328 {
329 name = g_utf8_strdown (name, len);
330 g_free (tmp);
331 tmp = name;
332 len = -1;
333 }
334

336 if (!contains_non_ascii (name, len))

356 * angstrom -> capital A with ring). So we have to lowercasify a
357 * second time. (This is more-or-less how the nameprep algorithm
358 * does it. If tolower(nfkc(tolower(X))) is guaranteed to be the
359 * same as tolower(nfkc(X)), then we could skip the first tolower,
360 * but I'm not sure it is.)

381 * characters, but glib does not appear to have that information.
382 */

681/**
682 * g_hostname_is_ip_address:
683 * @hostname: a hostname (or IP address in string form)
684 *
685 * Tests if @hostname is the string form of an IPv4 or IPv6 address.
686 * (Eg, "192.168.0.1".)
687 *
688 * Since 2.66, IPv6 addresses with a zone-id are accepted (RFC6874).
689 *
690 * Returns: %TRUE if @hostname is an IP address
691 *
692 * Since: 2.22
693 **/
694gboolean
695g_hostname_is_ip_address (const gchar *hostname)
696{
697 gchar *p, *end;
698 gint nsegments, octet;
699

701 * equivalent of that requires linking against winsock, so we just
702 * figure this out ourselves. Tested by tests/hostutils.c.
703 */
704
705 p = (char *)hostname;

707 if (strchr (p, ':'))

712 * IP address at all). This consists of eight ':'-separated
713 * segments, each containing a 1-4 digit hex number, except that
714 * optionally: (a) the last two segments can be replaced by an
715 * IPv4 address, and (b) a single span of 1 to 8 "0000" segments
716 * can be replaced with just "::".

720 skipped = FALSE(0);

724 * (We also handle half of the "string starts with ::" case

728 {

732 }

735 if (*p == ':' && !skipped)
736 {
737 skipped = TRUE(!(0));
738 nsegments++;
739
740 /* Handle the "string ends with ::" case */
741 if (!p[1])
742 p++;

744 continue;
745 }
746
747 /* Read the segment, make sure it's valid. */
748 for (end = p; g_ascii_isxdigit (*end)((g_ascii_table[(guchar) (*end)] & G_ASCII_XDIGIT) != 0); end++)
749 ;
750 if (end == p || end > p + 4)

752
753 if (*end == '.')
754 {

756 goto parse_ipv4;

770 /* Parse IPv4: N.N.N.N, where each N <= 255 and doesn't have leading 0s. */
771 for (nsegments = 0; nsegments < 4; nsegments++)
772 {
773 if (nsegments != 0)

776 return FALSE(0);

778 }
779
780 /* Check the segment; a little tricker than the IPv6 case since
781 * we can't allow extra leading 0s, and we can't assume that all
782 * strings of valid length are within range.
783 */

785 if (*p == '0')

787 else
788 {

796 }

798 return FALSE(0);
799
800 p = end;
801 }
802
803 /* If there's nothing left to parse, then it's ok. */
804 return !*p;
